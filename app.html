Passo 1

Leitura da Planilha

O usuario seleciona um arquivo .xlsx, .xls ou .csv. O FileReader le o binario e o SheetJS converte para um array de objetos JavaScript — cada objeto e uma linha da planilha. 

// O FileReader le o arquivo como binario (ArrayBuffer)
reader.readAsArrayBuffer(pendingFile);

reader.onload = (e) => {
  // SheetJS converte o binario em um workbook
  const data = new Uint8Array(e.target.result);
  const workbook = XLSX.read(data, { type: 'array' });

  // Pega a primeira aba da planilha
  const firstSheet = workbook.Sheets[workbook.SheetNames[0]];

  // Converte para array de objetos JS
  // Cada objeto = uma linha, cada chave = uma coluna
  const json = XLSX.utils.sheet_to_json(firstSheet, { defval: '' });

  // Resultado:
  // [
  //   { id: 1, prioridade: "verde", descricao: "Tarefa 123" },
  //   { id: 2, prioridade: "azul",  descricao: "Tarefa 456" },
  //   ...
  // ]

  rawData = json;
  headers = Object.keys(json[0]); // ["id", "prioridade", "descricao"]
};
Passo 2

Extracao de Valores Unicos

Quando o usuario clica em uma coluna, o sistema percorre todas as linhas e coleta os valores sem repetir, preservando a ordem de aparicao. Usa um Set para garantir unicidade. 

function selectColumn(col) {
  selectedColumn = col;

  // Set para rastrear o que ja foi visto
  const seen = new Set();
  uniqueValues = [];

  // Percorre cada linha da planilha
  rawData.forEach((row) => {
    const val = String(row[col] ?? '');

    // Se o valor ainda nao apareceu, adiciona a lista
    if (!seen.has(val)) {
      seen.add(val);
      uniqueValues.push(val);
    }
  });

  // Exemplo com coluna "prioridade":
  // uniqueValues = ["verde", "azul", "turquesa"]
  //
  // Esses valores sao exibidos na lista arrastavel
  // para o usuario definir a ordem que deseja.
}
Passo 3

Ordenacao Personalizada

O coracao do sistema. Cria um mapa de prioridade (Map) a partir da ordem definida pelo usuario. Depois usa Array.sort() comparando os indices — quem tem indice menor vem primeiro. Valores nao mapeados vao para o final. 

function generateResult() {
  // 1. Le a ordem que o usuario definiu arrastando os itens
  const items = sortableEl.querySelectorAll('li');

  // 2. Cria o mapa de prioridade:
  //    "azul"     => 0 (primeiro)
  //    "verde"    => 1 (segundo)
  //    "turquesa" => 2 (terceiro)
  const priorityMap = new Map();
  items.forEach((li, i) => {
    priorityMap.set(li.dataset.value, i);
  });

  // 3. Valores desconhecidos vao para o final
  const fallback = items.length;

  // 4. Ordena a planilha inteira usando o mapa
  sortedData = [...rawData].sort((a, b) => {
    const va = String(a[selectedColumn] ?? '');
    const vb = String(b[selectedColumn] ?? '');

    // Busca o indice de cada valor no mapa
    const pa = priorityMap.has(va)
      ? priorityMap.get(va) : fallback;
    const pb = priorityMap.has(vb)
      ? priorityMap.get(vb) : fallback;

    // Menor indice vem primeiro
    // Ex: "azul"(0) - "verde"(1) = -1 → azul vem antes
    return pa - pb;
  });
}
Passo 4

Exportacao do Resultado

Com os dados ordenados, o SheetJS converte de volta para uma planilha e gera o download. Tambem suporta .csv com BOM UTF-8 para garantir que acentos aparecam corretamente no Excel. 

// Download como .xlsx
function downloadXlsx() {
  // Converte o array de objetos de volta para uma sheet
  const ws = XLSX.utils.json_to_sheet(sortedData, {
    header: headers
  });

  // Cria um workbook e adiciona a sheet
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, 'Ordenado');

  // Dispara o download
  XLSX.writeFile(wb, 'planilha_customsort.xlsx');
}

// Download como .csv (com BOM UTF-8)
function downloadCsv() {
  const ws = XLSX.utils.json_to_sheet(sortedData, {
    header: headers
  });
  const csv = XLSX.utils.sheet_to_csv(ws);

  // BOM (\uFEFF) garante que o Excel reconheca UTF-8
  const blob = new Blob(
    ['\uFEFF' + csv],
    { type: 'text/csv;charset=utf-8;' }
  );

  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'planilha_customsort.csv';
  a.click();
  URL.revokeObjectURL(url);
}
